proc Dijkstra1(V, E, δ, s)

dist[v] = +∞ for all v ∈ V \ {s}

dist[s] = 0
Insert(Q,⟨dist[s], s⟩)

while Q ̸= ∅ do
  ⟨d, u⟩ = ExtractMin(Q)
  for (u, v) ∈ E ∩ ({u} × V ) do
    if dist[u] + δ(u, v) < dist[v] then
      dist[v] = dist[u] + δ(u, v)
      if v ∈ Q then
        DecreaseKey(Q, v, dist[v])
      else
        Insert(Q,⟨v, dist[v]⟩)

return dist
